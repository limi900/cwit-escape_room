    function diffuseBomb(wireSequence):
        # Correct sequence of wires to cut
        correctSequence = ["red", "green", "blue"]
        
        # Check if the input sequence matches the correct sequence
        for i in range(len(correctSequence)):
            if wireSequence[i] != correctSequence[i]:
                print("ERROR: Incorrect wire sequence! Bomb detonated.")
                return False
        
        # If the sequence is correct
        print("Success! Bomb diffused.")
        return True
